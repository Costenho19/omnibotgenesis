"""
OMNIX V5.2 QUANTUM ULTIMATE - Trading Service Enterprise (Simplified)
Versión simplificada para Railway con Kraken API
"""

import os
import logging
import hashlib
import hmac
import time
import base64
import urllib.parse
from typing import Dict, Any, Optional, List

logger = logging.getLogger(__name__)

try:
    import requests
    REQUESTS_AVAILABLE = True
except ImportError:
    REQUESTS_AVAILABLE = False
    logger.warning("⚠️ requests no disponible")


class TradingServiceEnterprise:
    """
    Trading Service Enterprise - Conexión con Kraken API
    Versión simplificada para Railway
    """
    
    def __init__(self):
        """Inicializar servicio de trading"""
        self.api_key = os.environ.get('KRAKEN_API_KEY')
        self.api_secret = os.environ.get('KRAKEN_API_SECRET')
        self.api_url = 'https://api.kraken.com'
        self.connected = False
        
        if not REQUESTS_AVAILABLE:
            logger.error("❌ requests no disponible")
            return
        
        if not self.api_key or not self.api_secret:
            logger.warning("⚠️ Kraken API keys no configuradas")
            return
        
        # Verificar conexión
        try:
            balance = self.get_balance()
            if balance:
                self.connected = True
                logger.info("✅ TRADING SERVICE ENTERPRISE CONECTADO A KRAKEN")
        except Exception as e:
            logger.error(f"❌ Error conectando a Kraken: {e}")
    
    def _sign_request(self, url_path: str, data: Dict) -> Dict[str, str]:
        """Firmar request para Kraken API"""
        try:
            post_data = urllib.parse.urlencode(data)
            encoded = (str(data['nonce']) + post_data).encode()
            message = url_path.encode() + hashlib.sha256(encoded).digest()
            
            signature = hmac.new(
                base64.b64decode(self.api_secret),
                message,
                hashlib.sha512
            )
            sig_digest = base64.b64encode(signature.digest())
            
            return {
                'API-Key': self.api_key,
                'API-Sign': sig_digest.decode()
            }
        except Exception as e:
            logger.error(f"❌ Error firmando request: {e}")
            return {}
    
    def get_balance(self) -> Optional[Dict[str, float]]:
        """Obtener balance de Kraken"""
        if not REQUESTS_AVAILABLE or not self.api_key:
            return None
        
        try:
            url_path = '/0/private/Balance'
            data = {'nonce': str(int(time.time() * 1000))}
            headers = self._sign_request(url_path, data)
            
            response = requests.post(
                self.api_url + url_path,
                headers=headers,
                data=data,
                timeout=10
            )
            
            result = response.json()
            if result.get('error'):
                logger.error(f"❌ Kraken error: {result['error']}")
                return None
            
            return result.get('result', {})
        except Exception as e:
            logger.error(f"❌ Error obteniendo balance: {e}")
            return None
    
    def get_ticker(self, pair: str = 'XBTUSD') -> Optional[Dict[str, Any]]:
        """Obtener precio actual de un par"""
        if not REQUESTS_AVAILABLE:
            return None
        
        try:
            url = f"{self.api_url}/0/public/Ticker?pair={pair}"
            response = requests.get(url, timeout=10)
            result = response.json()
            
            if result.get('error'):
                logger.error(f"❌ Kraken error: {result['error']}")
                return None
            
            data = result.get('result', {})
            return data.get(pair, data.get('XXBTZUSD', {}))
        except Exception as e:
            logger.error(f"❌ Error obteniendo ticker: {e}")
            return None
    
    def get_btc_price(self) -> Dict[str, Any]:
        """Obtener precio de Bitcoin con detalles"""
        ticker = self.get_ticker('XBTUSD')
        
        if not ticker:
            return {
                'price': 0.0,
                'change': 0.0,
                'volume': 0.0,
                'low_24h': 0.0,
                'high_24h': 0.0,
                'exchange': 'Kraken'
            }
        
        try:
            current_price = float(ticker['c'][0])
            open_price = float(ticker['o'])
            change_24h = ((current_price - open_price) / open_price) * 100
            
            return {
                'price': current_price,
                'change': change_24h,
                'volume': float(ticker['v'][1]),
                'low_24h': float(ticker['l'][1]),
                'high_24h': float(ticker['h'][1]),
                'exchange': 'Kraken'
            }
        except Exception as e:
            logger.error(f"❌ Error procesando ticker: {e}")
            return {
                'price': 0.0,
                'change': 0.0,
                'volume': 0.0,
                'low_24h': 0.0,
                'high_24h': 0.0,
                'exchange': 'Kraken'
            }
    
    def place_order(
        self,
        pair: str,
        side: str,
        amount: float,
        order_type: str = 'market'
    ) -> Optional[Dict[str, Any]]:
        """Colocar orden en Kraken"""
        if not REQUESTS_AVAILABLE or not self.api_key:
            return None
        
        try:
            url_path = '/0/private/AddOrder'
            data = {
                'nonce': str(int(time.time() * 1000)),
                'ordertype': order_type,
                'type': side,
                'volume': str(amount),
                'pair': pair
            }
            
            headers = self._sign_request(url_path, data)
            
            response = requests.post(
                self.api_url + url_path,
                headers=headers,
                data=data,
                timeout=10
            )
            
            result = response.json()
            if result.get('error'):
                logger.error(f"❌ Kraken order error: {result['error']}")
                return None
            
            logger.info(f"✅ Orden ejecutada: {side} {amount} {pair}")
            return result.get('result', {})
        except Exception as e:
            logger.error(f"❌ Error colocando orden: {e}")
            return None
    
    def health_check(self) -> Dict[str, Any]:
        """Verificar estado del servicio"""
        return {
            'service': 'TradingServiceEnterprise',
            'requests_available': REQUESTS_AVAILABLE,
            'connected': self.connected,
            'api_configured': bool(self.api_key and self.api_secret)
        }


# Instancia global
_trading_service_instance = None

def get_trading_service() -> TradingServiceEnterprise:
    """Obtener instancia singleton del servicio de trading"""
    global _trading_service_instance
    if _trading_service_instance is None:
        _trading_service_instance = TradingServiceEnterprise()
    return _trading_service_instance
