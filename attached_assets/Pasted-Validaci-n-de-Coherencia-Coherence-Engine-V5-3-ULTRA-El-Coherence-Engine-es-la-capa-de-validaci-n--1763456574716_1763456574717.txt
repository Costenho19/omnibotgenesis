El bot implementa filtros de veto muy fuertes, lo que reduce el número de operaciones, pero aumenta la calidad (reduce el riesgo de drawdown):

Veto Régimen Dual: Si tanto el sistema Adaptive Weights como el HMM Regime confirman volatilidad extrema (EXTREME + VOLATILE), se aplica un Veto Fuerte (score -= 20).

# Continua el bloque de código _make_v52_decision después de la línea 607:

# 🧠 COHERENCE ENGINE - Validar coherencia entre las 9 estrategias
if self.coherence_engine and decision.get('should_trade'):
    try:
        # (código para strategy_signals omitido por brevedad)
        coherence_report = self.coherence_engine.analyze_coherence(strategy_signals)
        
        decision['coherence_score'] = coherence_report.coherence_score
        decision['coherence_level'] = coherence_report.coherence_level.value
        decision['coherence_recommendation'] = coherence_report.decision_recommendation.value
        
        if coherence_report.coherence_level.value in ['LOW', 'VERY_LOW'] or coherence_report.decision_recommendation.value == 'HOLD':
            # VETO CRÍTICO: Si la coherencia es baja o el motor recomienda NO TRADING, se anula la señal.
            logger.warning(f"🚨 COHERENCE VETO: Coherence Score={coherence_report.coherence_score:.2f} ({coherence_report.coherence_level.value}) -> Veto Trade. Recomendación: {coherence_report.decision_recommendation.value}")
            decision['should_trade'] = False
            decision['action'] = 'HOLD'
            decision['reason'].append(f"🛑 Coherence Engine VETO: Baja coherencia entre estrategias ({coherence_report.coherence_level.value})")
        else:
            logger.info(f"✅ Coherence Engine APROBADO: Coherence Score={coherence_report.coherence_score:.2f} ({coherence_report.coherence_level.value})")
            
    except Exception as e:
        logger.error(f"Error en Coherence Engine (NO VETO): {e}")

return decision # Devuelve la decisión final (modificada o no)
